package comccc.example.user.linkadhaar;

import android.content.Context;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.BaseAdapter;
import android.widget.TextView;

 * Created by USER on 23-09-2017.
 */

public class Attendance_adapter extends BaseAdapter {

    Context context;
    LayoutInflater inflater;
    Attendance_adapter.ViewHolder holder;
    public static Attendance_details table1;


    public Attendance_adapter(Context context) {
        this.context = context;
        inflater=(LayoutInflater)context.getSystemService(Context.LAYOUT_INFLATER_SERVICE);
    }

    @Override
    public int getCount()
    {
        return Attendance.detaillist.size();
    }

    @Override
    public Object getItem(int i)
    {
        return Attendance.detaillist.get(i);
    }

    @Override
    public long getItemId(int i) {

        return i;
    }

    @Override
    public View getView(int i, View view, ViewGroup viewGroup) {

        if(view == null){
            holder = new Attendance_adapter.ViewHolder();
            view = inflater.inflate(R.layout.attendance_list,null);
            holder.text1 =(TextView)view.findViewById(R.id.sub1);
            holder.text2 =(TextView)view.findViewById(R.id.present1);
            holder.text3 =(TextView)view.findViewById(R.id.periods1);
            holder.text4 =(TextView)view.findViewById(R.id.percent1);

            // view.setTag(holder);
        }else
        {
            holder = (Attendance_adapter.ViewHolder)view.getTag();
        }

        table1 = (Attendance_details) getItem(i);

      /*  holder.text1.setTag(table);
        holder.text2.setTag(table);
        holder.text3.setTag(table);
        holder.text4.setTag(table);
        holder.text5.setTag(table);*/


        holder.text1.setText(String.valueOf(table1.getSubject()));
        holder.text2.setText(String.valueOf(table1.getTpresents()));
        holder.text3.setText(String.valueOf(table1.getTperiods()));
        holder.text4.setText(String.valueOf(table1.getPercentage()));
       // holder.text5.setText(String.valueOf(table1.getSection()));

        return view;

    }


    public static class ViewHolder{
        TextView text1,text2,text3,text4;
    }
    }
